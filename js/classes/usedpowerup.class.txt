class usedPowerUp{
	constructor(name, target, icon){
		this.name = name; //Name of the powerup. Useful for activation prompts 
		this.target = target;
		this.icon = icon;
	}

	getName(){
		return this.name;
	}

	getIcon(){
		return this.icon;
	}

	getTarget(){
		return this.target;
	}

}
